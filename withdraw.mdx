---
title: "Build withdraw (yield)"
api: "POST /yield/withdraw"
description: "Build an unsigned withdraw transaction with fee_policy metadata derived from tracked principal and on-chain earnings."
authMethod: "key"
---

<ParamField header="x-api-key" type="string" required placeholder="your-api-key">
  Your StarkNode API key. You can find it in your [dashboard](https://app.starknode.io).
</ParamField>

<ParamField body="provider" type="enum<string>" required>
  `jupiter_lend` or `kamino`.
</ParamField>

<ParamField body="chain_name" type="string" required>
  Enabled chain, e.g. `solana`.
</ParamField>

<ParamField body="asset" type="string" required>
  Token symbol.
</ParamField>

<ParamField body="wallet" type="string" required>
  User wallet public key.
</ParamField>

<ParamField body="amount" type="string" required>
  Amount to withdraw (validated against tracked and protocol balance).
</ParamField>

### Response

<ResponseField name="data.position_id" type="string" required>
  Snapshot id for [Broadcast](/api-reference/yield/broadcast) with `operation: "withdraw"`.
</ResponseField>

<ResponseField name="data.fee_policy" type="object" required>
  Fee application summary for this withdraw build.
</ResponseField>

<ResponseField name="data.transaction" type="string" required>
  Base64 unsigned Solana wire.
</ResponseField>
