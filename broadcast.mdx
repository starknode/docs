---
title: "Broadcast yield operation"
api: "POST /api/yield/broadcast"
description: "Submit signed Solana transaction(s) for deposit, withdraw, or rebalance. StarkNode broadcasts and updates positions."
authMethod: "key"
---

<ParamField header="x-api-key" type="string" required placeholder="your-api-key">
  Your StarkNode API key. You can find it in your [dashboard](https://app.starknode.io).
</ParamField>

<ParamField body="operation" type="enum<string>" required>
  `deposit`, `withdraw`, or `rebalance`.
</ParamField>

<ParamField body="op_signed" type="string" required>
  Base64 signed Solana wire, or JSON array of two base64 strings for rebalance two-step mode. Deposit and withdraw: exactly **one** string. Rebalance: **one** string (merged) or **array of two** strings `[withdraw, deposit]`.
</ParamField>

<ParamField body="position_id" type="string">
  Required for `deposit` and `withdraw` — id from the build response.
</ParamField>

<ParamField body="position_out" type="string">
  Required for `rebalance` — `position_out_id` from rebalance build.
</ParamField>

<ParamField body="position_in" type="string">
  Required for `rebalance` — `position_in_id` from rebalance build.
</ParamField>

### Response

<ResponseField name="data.status" type="number">
  Receipt-style status from broadcast (`1` when confirmed successful on Solana).
</ResponseField>

<ResponseField name="data.transactionHash" type="string">
  On-chain signature or hash when available (deposit/withdraw; may be empty for some rebalance shapes).
</ResponseField>

<ResponseField name="data.position_in" type="object">
  Present for deposit and rebalance confirmations.
</ResponseField>

<ResponseField name="data.position_out" type="object">
  Present for withdraw and rebalance confirmations.
</ResponseField>
