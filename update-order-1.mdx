```javascript Node.js
const orderId = "clxxxxxxxxxxxxxxxxxxxxxxxx";
const response = await fetch(
```

```python Python
import requests

order_id = "clxxxxxxxxxxxxxxxxxxxxxxxx"
response = requests.patch(
    f"https://api.starknode.io/api/order/{order_id}",

    json={"amount_from": "150.00"},
)
data = response.json()
```
